<script>

let appData = {};

let selectedIncomeTax = 0;

const RESIDENCE_TAX = 0.1; // 住民税 10%


const GAS_URL = "https://script.google.com/macros/s/AKfycbxvKlbO0KCBNeVgiIPD_lSyFYrudMWmrPumzkocR6y7ArABYVuwwgKPUjgDqJFP9M9C/exec"; 

window.onload = () => {

 fetch(GAS_URL)

 .then(response => response.json())

 .then(data => {

 appData = data;

 initApp();

 });

 };


function initApp() {

  const incomeSel = document.getElementById('incomeTaxSelect');

  appData.taxes.income.forEach((v, i) => {

    let opt = new Option(v + "%", v);

    if(v == 20) opt.selected = true;

    incomeSel.add(opt);

  });

  if (incomeSel.selectedIndex === -1 && incomeSel.options.length > 0) incomeSel.options[0].selected = true;

  

  const linkCont = document.getElementById('linkContainer');

  appData.links.forEach(l => {

    const a = document.createElement('a');

    a.href = l.url; a.className = 'menu-link'; a.target = '_blank';

    a.innerText = l.title;

    linkCont.appendChild(a);

  });

  saveSettings();

  addRouteRow();

}


function addRouteRow() {

  const id = Date.now();

  const optionsHtml = appData.airports.map(a => {

    if (a.isCategory) {

      return `<option value="${a.name}" disabled style="background:#eee; font-weight:bold; color:#E60012; padding:5px 0;">${a.name}</option>`;

    } else {

      return `<option value="${a.name}">${a.name}</option>`;

    }

  }).join('');


  const html = `

    <div class="route-card" id="row-${id}">

      <select id="dep-${id}" onchange="updateArrivalOptions('${id}')">

        <option value="">出発地</option>

        ${optionsHtml}

      </select>

      <span class="material-icons arrow-icon">arrow_forward</span>

      <select id="arr-${id}" onchange="calculate()">

        <option value="">到着地</option>

        ${optionsHtml}

      </select>

      <span class="material-icons" style="color:#ddd; cursor:pointer; font-size:26px; margin-left:4px;" onclick="removeRow('${id}')">cancel</span>

    </div>

  `;

  document.getElementById('routeContainer').insertAdjacentHTML('beforeend', html);

}


function updateArrivalOptions(id) {

  const depVal = document.getElementById(`dep-${id}`).value;

  const arrSelect = document.getElementById(`arr-${id}`);

  const possibleArrs = appData.routes

    .filter(r => r.name.includes(depVal))

    .map(r => r.name.replace(depVal, "").replace("=", ""));


  Array.from(arrSelect.options).forEach(opt => {

    if (opt.value === "" || opt.disabled) return;

    opt.disabled = !possibleArrs.includes(opt.value);

    opt.style.color = opt.disabled ? "#ccc" : "#000";

  });

  calculate();

}


function removeRow(id) {

  if (document.querySelectorAll('.route-card').length > 1) { 

    document.getElementById(`row-${id}`).remove(); 

    calculate(); 

  }

}


function saveSettings() {

  const val = document.getElementById('incomeTaxSelect').value;

  selectedIncomeTax = parseFloat(val) / 100;

  document.getElementById('taxHeader').innerText = `【所得税率】${val}% 【住民税率(所得割)】10%`;

  closeSettings();

  calculate();

}


/**

 * 運賃計算メインロジック

 */

function calculate() {

  let totalLT00 = 0, totalLTZL = 0;

  const adults = parseInt(document.getElementById('adultCount').value) || 0;

  const children = parseInt(document.getElementById('childCount').value) || 0;

  let detailHtml = "";


  document.querySelectorAll('.route-card').forEach(row => {

    const id = row.id.split('-')[1];

    const dep = document.getElementById(`dep-${id}`).value;

    const arr = document.getElementById(`arr-${id}`).value;

    

    if (dep && arr) {

      const route = appData.routes.find(r => r.name === `${dep}=${arr}` || r.name === `${arr}=${dep}`);

      const dPort = appData.airports.find(a => a.name === dep && !a.isCategory);

      const aPort = appData.airports.find(a => a.name === arr && !a.isCategory);

      

      if (route && dPort && aPort) {

        // --- 大人料金の計算 ---

        const fareA = route.fareA;

        const pfcSumA = dPort.pfcA + aPort.pfcA;

        const lt00A = (fareA * selectedIncomeTax) + (fareA * RESIDENCE_TAX) + pfcSumA;

        const ltzlA = fareA + pfcSumA;


        // --- 子供料金の計算 ---

        const fareC = route.fareC;

        const pfcSumC = dPort.pfcC + aPort.pfcC;

        const lt00C = (fareC * selectedIncomeTax) + (fareC * RESIDENCE_TAX) + pfcSumC;

        const ltzlC = fareC + pfcSumC;


        // 合計への加算

        totalLT00 += (lt00A * adults) + (lt00C * children);

        totalLTZL += (ltzlA * adults) + (ltzlC * children);

        

        detailHtml += `

          <div style="border-bottom:1px solid #eee; padding:12px 0;">

            <b style="color:var(--primary)">${dep} ⇄ ${arr}</b><br>

            <span style="font-size:12px; color:var(--gray)">

              大人: LT00 ¥${Math.round(lt00A).toLocaleString()} / LTZL ¥${Math.round(ltzlA).toLocaleString()}<br>

              子供: LT00 ¥${Math.round(lt00C).toLocaleString()} / LTZL ¥${Math.round(ltzlC).toLocaleString()}

            </span>

          </div>`;

      }

    }

  });

  document.getElementById('totalLT00').innerText = `¥${Math.round(totalLT00).toLocaleString()}`;

  document.getElementById('totalLTZL').innerText = `¥${Math.round(totalLTZL).toLocaleString()}`;

  document.getElementById('breakdownSection').innerHTML = detailHtml || "路線を選択してください";

}


function toggleMenu() {

  document.getElementById('menuDrawer').classList.toggle('active');

  const overlay = document.getElementById('menuOverlay');

  overlay.style.display = (overlay.style.display === 'block') ? 'none' : 'block';

}

function openSettings() { document.getElementById('settingsModal').style.display = 'block'; }

function closeSettings() { document.getElementById('settingsModal').style.display = 'none'; }

function toggleBreakdown() {

  const b = document.getElementById('breakdownSection');

  const icon = document.getElementById('expandIcon');

  const isShow = (b.style.display === 'block');

  b.style.display = isShow ? 'none' : 'block';

  icon.innerText = isShow ? 'add_circle' : 'remove_circle';

}
